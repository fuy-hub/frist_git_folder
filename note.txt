git 是個可以追蹤log控制checkout檔案版本、變更的系統
git --version // 看git是否安裝
git init // 初始化建立.git檔紀錄訊息
git add 檔案 // 追蹤
git add . // 追蹤track此目錄所有檔
git commit -m "還原點名稱" // 建立還原點(提交所有追蹤檔案狀態)
git stutas // 當前目錄，檔案狀態，會顯示未追蹤檔
git log // 列出詳細的提交歷史 輸入q跳出
git log --online // 列出所有的提交歷史(簡化)
git diff 還原點ID -- 檔案名稱 
//比較不同 -還原點時那行訊息 +現在那行修改的地方
git checkout 還原點ID -- 檔案名稱 
// 將檔案還原到還原點狀態
git reset --hard 還原點ID 
// 當前目錄所有檔，還原到還原點ID版本，刪除之後提交記錄 
.gitignore // 設定忽視不提交的檔案
*.副檔名 // 有此副檔名的檔案
//1.連接儲存庫 2.建立主支 3.第一推送本機檔案
git remote add origin https://github.com/fuy-hub/frist_git_folder.git
git branch -M main
git push -u origin main

git push // 第一次之後推送
git clone 儲存庫網址 //協作者將儲存庫第一次專案拉至本機
git pull //協作者第一次之後將變更後儲存庫專案拉至本機
git branch //查看當前和所有分支
git checkout -b 分支名稱 //改變當前目錄分支
git push origin 分支名稱 //非主分支推送
